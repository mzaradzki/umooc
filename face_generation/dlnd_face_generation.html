Preprocess the Data¶

Since the project's main focus is on building the GANs, we'll preprocess the data for you. The values of the MNIST and CelebA dataset will be in the range of -0.5 to 0.5 of 28x28 dimensional images. The CelebA images will be cropped to remove parts of the image that don't include a face, then resized down to 28x28.

The MNIST images are black and white images with a single color channel while the CelebA images have 3 color channels (RGB color channel).

def discriminator(images, reuse=False):
    """
    Create the discriminator network
    :param image: Tensor of input image(s)
    :param reuse: Boolean if the weights should be reused
    :return: Tuple of (tensor output of the discriminator, tensor logits of the discriminator)
    """
    # TODO: Implement Function
    with tf.variable_scope('discriminator', reuse=reuse):
        if False: # First version, non convolutional
            n_units = 128
            # Hidden layer
            shape = images.get_shape().as_list()
            dim = np.prod(shape[1:])
            x = tf.reshape(images, (-1, dim))
            
            h1 = tf.layers.dense(x, n_units, activation=None)
            h1 = leaky_relu(h1)

            logits = tf.layers.dense(h1, 1, activation=None)
            out = tf.sigmoid(logits)

            return out, logits
        else:
            # Image shape = 28x28x3
            x1 = tf.layers.conv2d(images, 64, kernel_size, strides=2, padding='same')
            # WARNING : no batchnorm at first layer
            x1 = leaky_relu(x1)
            # new shape = 14x14x64

            x2 = tf.layers.conv2d(x1, 128, kernel_size, strides=2, padding='same')
            x2 = tf.layers.batch_normalization(x2, training=True)
            x2 = leaky_relu(x2)
            # new shape = 7x7x128

            x3 = tf.layers.conv2d(x2, 256, kernel_size, strides=2, padding='same')
            x3 = tf.layers.batch_normalization(x3, training=True)
            x3 = leaky_relu(x3)
            # new shape = 4x4x256

            flat = tf.reshape(x3, (-1, 4*4*256))
            logits = tf.layers.dense(flat, 1)
            out = tf.sigmoid(logits)

            return out, logits

def generator(z, out_channel_dim, is_train=True):
    """
    Create the generator network
    :param z: Input z
    :param out_channel_dim: The number of channels in the output image
    :param is_train: Boolean if generator is being used for training
    :return: The tensor output of the generator
    """
    # TODO: Implement Function
    with tf.variable_scope('generator', reuse=not(is_train)):
        if False: # First version, non convolutional
            n_units = 28*28*out_channel_dim
            # Hidden layer
            h1 = tf.layers.dense(z, n_units, activation=None)
            h1 = leaky_relu(h1)

            logits = tf.reshape(h1, (-1, 28, 28, out_channel_dim))

            # Logits and tanh output
            out = tf.tanh(logits) # WARNING : range [-1;+1]
            
            return out
        else:
            # First fully connected layer
            x0 = tf.layers.dense(z, 4*4*512)
            
            x1 = tf.reshape(x0, (-1, 4, 4, 512))
            x1 = tf.layers.batch_normalization(x1, training=is_train)
            x1 = leaky_relu(x1)
            # new shape = 4x4x512

            x2 = tf.layers.conv2d_transpose(x1, 256, kernel_size, strides=2, padding='same')
            x2 = tf.layers.batch_normalization(x2, training=is_train)
            x2 = leaky_relu(x2)
            # new shape = 8x8x256 now

            x3 = tf.layers.conv2d_transpose(x2, 128, kernel_size, strides=2, padding='same')
            x3 = tf.layers.batch_normalization(x3, training=is_train)
            x3 = leaky_relu(x3)
            # new shape = 16x16x128

            pre_logits = tf.layers.conv2d_transpose(x3, out_channel_dim, kernel_size, strides=2, padding='same')
            # new shape = 32x32x3
            
            # INFO : needs to crop borders
            logits = tf.slice(pre_logits, [0,2,2,0], [-1,28,28,-1])
            # new shape = 28x28x3
            
            out = tf.tanh(logits) # WARNING : range [-1;+1]

            return out

def model_loss(input_real, input_z, out_channel_dim):
    """
    Get the loss for the discriminator and generator
    :param input_real: Images from the real dataset
    :param input_z: Z input
    :param out_channel_dim: The number of channels in the output image
    :return: A tuple of (discriminator loss, generator loss)
    """
    # TODO: Implement Function
    
    g_model = generator(input_z, out_channel_dim, is_train=True) # WARNING : not 100% sure about True/False
    
    d_model_real, d_logits_real = discriminator(input_real, reuse=False) # WARNING : not 100% sure about True/False
    d_model_fake, d_logits_fake = discriminator(g_model, reuse=True) # WARNING : not 100% sure about True/False
    
    d_loss_real = tf.reduce_mean(
                        tf.nn.sigmoid_cross_entropy_with_logits(logits=d_logits_real, 
                                                              labels=tf.ones_like(d_logits_real) * (1 - smooth)),
                        name='d_loss_real') # INFO : this error means we aim to have reals labelled as 1s
    d_loss_fake = tf.reduce_mean(
                        tf.nn.sigmoid_cross_entropy_with_logits(logits=d_logits_fake, 
                                                              labels=tf.zeros_like(d_logits_real)),
                        name='d_loss_fake') # INFO : this error means we aim to have fakes labelled as 0s
    
    d_loss = d_loss_real + d_loss_fake

    g_loss = tf.reduce_mean(
                        tf.nn.sigmoid_cross_entropy_with_logits(logits=d_logits_fake,
                                                         labels=tf.ones_like(d_logits_fake)),
                        name='g_loss') # INFO : this error means we aim for G to fool D into labelling fakes as 1s
    
    return d_loss, g_loss

def model_opt(d_loss, g_loss, learning_rate, beta1):
    """
    Get optimization operations
    :param d_loss: Discriminator loss Tensor
    :param g_loss: Generator loss Tensor
    :param learning_rate: Learning Rate Placeholder
    :param beta1: The exponential decay rate for the 1st moment in the optimizer
    :return: A tuple of (discriminator training operation, generator training operation)
    """
    # TODO: Implement Function
    t_vars = tf.trainable_variables()
    
    g_vars = [var for var in t_vars if var.name.startswith('generator')]
    d_vars = [var for var in t_vars if var.name.startswith('discriminator')]

    d_train_opt = tf.train.AdamOptimizer(learning_rate, beta1).minimize(d_loss, var_list=d_vars)
    g_train_opt = tf.train.AdamOptimizer(learning_rate, beta1).minimize(g_loss, var_list=g_vars)
    
    return d_train_opt, g_train_opt

def show_generator_output(sess, n_images, input_z, out_channel_dim, image_mode):
    """
    Show example output for the generator

def train(epoch_count, batch_size, z_dim, learning_rate, beta1, get_batches, data_shape, data_image_mode):
    """
    Train the GAN
    :param epoch_count: Number of epochs
    :param batch_size: Batch Size
    :param z_dim: Z dimension
    :param learning_rate: Learning Rate
    :param beta1: The exponential decay rate for the 1st moment in the optimizer
    :param get_batches: Function to get batches
    :param data_shape: Shape of the data
    :param data_image_mode: The image mode to use for images ("RGB" or "L")
    """
    # TODO: Build Model
    input_real, input_z, lr = model_inputs(data_shape[1], data_shape[2], data_shape[3], z_dim)
    d_loss, g_loss = model_loss(input_real, input_z, data_shape[3])
    d_train_opt, g_train_opt = model_opt(d_loss, g_loss, lr, beta1)
    
    losses = []
    
    with tf.Session() as sess:
        sess.run(tf.global_variables_initializer())
        for epoch_i in range(epoch_count):
            c = 0
            for batch_images in get_batches(batch_size):
                
                # INFO : to check if input has been transformed into [–1;+1] range
                # INFO : found out by printing that I needed to scale by 2 to fit range bounds
                if (2*batch_images.min() < -1):
                    raise ValueError('out of range (too low) input pixel value')
                if (2*batch_images.max()> 1):
                    raise ValueError('out of range (too high) input pixel value')
                
                batch_z = np.random.uniform(-1, 1, size=(batch_size, z_dim))
                
                # Run optimizers
                _ = sess.run(d_train_opt, feed_dict={input_real: batch_images*2,
                                                     input_z: batch_z,
                                                     lr: learning_rate})
                
                _ = sess.run(g_train_opt, feed_dict={input_z: batch_z,
                                                     lr: learning_rate})
                
                if (c % 100 == 0):
                    show_generator_output(sess,
                                          4,
                                          input_z,
                                          data_shape[3],
                                          data_image_mode)
                c+= 1
                
                
            # After each each epoch print the losses (remark: G loss should be < D loss)
            train_loss_d = sess.run(d_loss, {input_real: batch_images*2,
                                             input_z: batch_z})
            train_loss_g = g_loss.eval({input_z: batch_z})

            print("Epoch {}/{}...".format(epoch_i+1, epochs),
                  "Discriminator Loss: {:.4f}...".format(train_loss_d),
                  "Generator Loss: {:.4f}".format(train_loss_g))    
            
            # Save losses to view after training
            losses.append((train_loss_d, train_loss_g))
